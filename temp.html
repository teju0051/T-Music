#include <stdio.h>

void reverseArray(int *arr, int n) {
    int *start = arr;
    int *end = arr + n - 1;
    while (start < end) {
        int temp = *start;
        *start = *end;
        *end = temp;
        start++;
        end--;
    }
}

    int n = 8;

void multiplyMatrices(int (*a)[3], int (*b)[3], int (*result)[3]) {
    for (int i = 0; i < 3; i++) {

#include <stdio.h>

// Function to concatenate src string to dest string using pointers
void concatenate(char *dest, char *src) {
    // Move pointer to the end of dest string
    while (*dest != '\0') {
        dest++;
    }
    // Copy src string to dest
    while (*src != '\0') {
        *dest = *src;
        dest++;
        src++;
    }
    *dest = '\0'; // Null terminate the concatenated string
}

int main() {

    scanf("%[^\n]", str1);

void insertBeg(struct Node **head, int val) {
    struct Node *n = createNode(val);
    n->next = *head; *head = n;

// Insert at end
void insertEnd(struct Node **head, int val) {
    struct Node *n = createNode(val);
    if (!*head) { *head = n; return; }
    struct Node *temp = *head; while(temp->next) temp = temp->next;

#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node *next;
};

// Append node at end
void append(struct Node **head, int val) {
    struct Node *newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = val;
    if (!*head) { 
        *head = newNode; 
        newNode->next = *head; 
        return; 
    }
    struct Node *temp = *head;
    while (temp->next != *head) temp = temp->next;
    temp->next = newNode;
    newNode->next = *head;
}

// Add node at beginning
void addBeginning(struct Node **head, int val) {
    struct Node *newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = val;
    if (!*head) { 
        *head = newNode; 
        newNode->next = *head; 
        return; 
    }
    struct Node *temp = *head;
    while (temp->next != *head) temp = temp->next;
    newNode->next = *head;
    temp->next = newNode;
    *head = newNode;
}

// Count nodes
int count(struct Node *head) {
    if (!head) return 0;
    int c = 1;
    struct Node *temp = head;
    while (temp->next != head) { temp = temp->next; c++; }
    return c;
}

// Display list
void display(struct Node *head) {
    if (!head) { printf("List is empty\n"); return; }
    struct Node *temp = head;
    do {
        printf("%d -> ", temp->data);
        temp = temp->next;
    } while (temp != head);
    printf("(head)\n");
}

#include <stdio.h>

int stack[MAX];
int top = -1;

// Push operation
void push(int val) {
    if (top == MAX - 1) {
        printf("Stack Overflow!\n");
        return;
    }
    stack[++top] = val;
    printf("%d pushed to stack.\n", val);
}

// Pop operation
void pop() {
    if (top == -1) {
        printf("Stack Underflow!\n");
        return;
    }
    printf("%d popped from stack.\n", stack[top--]);
}

// Display stack
void display() {
    if (top == -1) {
        printf("Stack is empty.\n");
        return;
    }
    printf("Stack elements: ");
    for (int i = top; i >= 0; i--)
        printf("%d ", stack[i]);
    printf("\n");
}

#include <stdio.h>
#define MAX 100

int queue[MAX];
int front = -1, rear = -1;

// Enqueue operation
void enqueue(int val) {
    if (rear == MAX - 1) {
        printf("Queue Overflow!\n");
        return;
    }
    if (front == -1) front = 0; // First element
    queue[++rear] = val;
    printf("%d enqueued to queue.\n", val);
}

// Dequeue operation
void dequeue() {
    if (front == -1 || front > rear) {
        printf("Queue Underflow!\n");
        return;
    }
    printf("%d dequeued from queue.\n", queue[front++]);
}

// Display queue
void display() {
    if (front == -1 || front > rear) {
        printf("Queue is empty.\n");
        return;
    }
    printf("Queue elements: ");
    for (int i = front; i <= rear; i++)
        printf("%d ", queue[i]);
    printf("\n");
}

int main() {
    int choice, val;
    do {
        printf("\n1.Enqueue 2.Dequeue 3.Display 4.Exit\nEnter choice: ");
        scanf("%d", &choice);
        switch(choice) {
            case 1: 
                printf("Enter value: "); 
                scanf("%d", &val); 
                enqueue(val); 
                break;
            case 2: dequeue(); break;
            case 3: display(); break;
            case 4: printf("Exiting...\n"); break;

        }
