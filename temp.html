Program 4: Fibonacci Series

n = int(input("Enter number of terms: "))
a, b = 0, 1

while count < n:

Program 5: Sum of Digits

num = int(input("Enter a number: "))
sum_of_digits = 0
temp = num

while temp > 0:
    digit = temp % 10
    sum_of_digits += digit
    temp = temp // 10

print(f"The sum of digits of {num} is {sum_of_digits}")

Program 7: Array Aliasing and Copying

# Program 7: Array Aliasing and Copying

# Create an array (list) from user input
arr = list(map(int, input("Enter elements of the array separated by space: ").split()))

# Aliasing
alias_arr = arr  # aliasing (both point to same list)
alias_arr[0] = -1  # modifying alias modifies original

print("After aliasing modification:")
print("Original array:", arr)
print("Aliased array:", alias_arr)

# Copying
copy_arr = arr.copy()  # creates a shallow copy
copy_arr[1] = -2  # modifying copy does NOT affect original

print("\nAfter copying modification:")
print("Original array:", arr)
print("Copied array:", copy_arr)

Program 10: Recursive Function to Print Factorial

# Program 10: Recursive Factorial

def factorial(n):
    if n == 0 or n == 1:
        return 1
    else:
        return n * factorial(n - 1)

# Accept number from user
num = int(input("Enter a number to find factorial: "))
if num < 0:
    print("Factorial is not defined for negative numbers.")
else:
    print(f"Factorial of {num} is {factorial(num)}")

Program 12: Check if Two Lists Have at Least One Common Member

# Program 12: Check Common Elements in Two Lists

# Accept two lists from user
list1 = list(map(int, input("Enter elements of first list separated by space: ").split()))
list2 = list(map(int, input("Enter elements of second list separated by space: ").split()))

# Check for common elements
common = False
for item in list1:
    if item in list2:
        common = True
        break

print("Do the lists have at least one common member?", common)

Program 13: Count Occurrence of a Specific Word in a File Using Regex

# Program 13: Count Occurrence of a Specific Word in a File Using Regex
import re

# Accept filename and word from user
filename = input("Enter filename: ")
word_to_count = input("Enter the word to count: ")

try:
    with open(filename, 'r') as file:
        content = file.read()
        # Use regular expression to find all occurrences (case-insensitive)
        matches = re.findall(r'\b' + re.escape(word_to_count) + r'\b', content, re.IGNORECASE)
        print(f"The word '{word_to_count}' occurred {len(matches)} time(s) in the file.")
except FileNotFoundError:
    print(f"File '{filename}' not found.")
